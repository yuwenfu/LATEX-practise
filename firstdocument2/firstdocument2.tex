\documentclass{book} %article,report,letter

\usepackage{enumerate}     %用于修改前面的编号字体和编号大小写
%\usepackage{longtable}     %

\begin{document}
\begin{itemize}
  \item this is item 1.
  \begin{itemize}       %insert-Itemize子嵌套
    \item This is subitem 1.
    \item This is subitem 2.
    \item This is subitem 3.
  \end{itemize}
  \item this is item 2.
  \item this is item 3.
\end{itemize}

\begin{enumerate} [A]      %insert-enumerate
  \item This is item 1.
    \begin{enumerate}       %insert-enumerate子嵌套
    \item This is subitem 1.
    \item This is subitem 2.
    \item This is subitem 3.
  \end{enumerate}
  \item This is item 2.
  \item This is item 3.
\end{enumerate}
                            %使用enumerate宏包
\begin{enumerate} [\bfseries A.]      %使用标题标号为A.;\bfseries是粗体命令
\setcounter {enumi}{4}  %设定计数器--改变编号的起始字母，以第五个开始计数，enumii表示第二层
  \item This is item 1.
  %\begin{enumerate}[\Sffamily a.]       %使用标题标号为a. ;字体命令\sffamily
  \begin{enumerate}[a.] 
  \setcounter {enumii}{4}   %设定计数器--改变编号的起始字母，以第五个开始计数，enumii表示第二层
    \item This is subitem 1.
    \item This is subitem 2.
    \item This is subitem 3.
  \end{enumerate}
  \item This is item 2.
  \item This is item 3.
\end{enumerate}
